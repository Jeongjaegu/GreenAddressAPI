Simple Python full access login example¶

Here we demonstrate how you can implement a simple Python application which logs in to GreenAddress wallet with full control, using mnemonic passphrase, generate a subaccount and log out.

Requirements¶

To implement this example, we use the following libraries, so make sure you install them first:
  • Twisted: pip install twisted
  • pyOpenSSL: pip install pyopenssl
  • service_identity: pip install service-identity
  • Autobahn|Python 0.8.12: pip install autobahn==0.8.12
  • pycoin 0.42: pip install pycoin==0.42
  • Requests 2.3.0: pip install requests==2.3.0
  • mnemonic: pip install mnemonic

Using the GreenAddress WAMP server¶

Retrieving a token¶

To connect to GreenAddress server, you need to acquire a token over HTTP which is used for protecting our servers against Denial of Service attacks:

import requests
token = requests.get('https://greenaddress.it/token/').text

Full control login, interactive example¶

In this example we use the mnemonic passphrase for a full control login. In full control login all the features are unlocked (i.e. you can’t create subaccounts in watch-only mode, and more).

Look at the interactive example below, and try it, to see how simple is passing and receiving data from GA servers.

try:
    from autobahn.wamp import WampClientFactory, WampCraClientProtocol
    from autobahn.websocket import connectWS
except:
    from autobahn.wamp1.protocol import WampClientFactory, WampCraClientProtocol
    from autobahn.twisted.websocket import connectWS

# Imports could change between Autobahn framework versions.

from twisted.internet import reactor
import mnemonic, random, binascii, struct, requests
from pycoin.key import bip32, Key
from pycoin.serialize import h2b
from pycoin import ecdsa as pycoin_ecdsa
from getpass import getpass

class GreenAddressClientProtocol(WampCraClientProtocol):
    def __init__(self, token):
        self.token = token
        self.mnemonic_phrase = None

    def onSessionOpen(self):
        print 'Welcome to GreenAddress mnemonic login example'
        print '\nThis script will login to GA in full mode'
        self.mnemonic_phrase = getpass('Plese write your mnemonic phrase and presse enter: ')
        self._prefix = ('gait', 'http://greenaddressit.com/')
        self.prefix(self._prefix[0], self._prefix[1])
        d = self.authenticate(authKey = self.token,
                            authExtra = None,
                            authSecret = self.token)
        d.addCallbacks(self.onAuthSuccess, self.onAuthError)

    def onClose(self, wasClean, code, reason):
        print 'Closing session'
        reactor.stop()

    def onAuthSuccess(self, permissions):
        print "\nAuthentication Success!", permissions
        self.mnemonicLogin(self.mnemonic_phrase)

    def onAuthError(self, e):
        uri, desc, details = e.value.args
        print "Authentication Error!", uri, desc, details

    def mnemonicLogin(self, mnemonic_phrase):
        def _onChallenge(challenge):
            checkforce = True # Force logout on already logged in session.
            signature = pycoin_ecdsa.sign(pycoin_ecdsa.generator_secp256k1, wallet_login.secret_exponent, int(challenge))
            d = self.call("gait:login/authenticate", signature, checkforce, path)
            d.addCallback(self.onLogin)
            d.addErrback(lambda failure: failure.printTraceback())

        # 1. Master wallet
        seed = mnemonic.Mnemonic.to_seed(mnemonic_phrase)
        self.wallet = bip32.Wallet.from_master_secret(seed, 'BTC')
        # Change 'BTC' to 'XTN' for Testnet

        # 2. Login wallet
        path = '%X' % random.randint(0, 2**64-1)
        while len(path) < 16: path = '0' + path
        path_bin = h2b(path)
        path_str = '/'.join(str(struct.unpack('!H', path_bin[i*2:(i+1)*2])[0]) for i in xrange(4))
        wallet_login = self.wallet.subkey_for_path(path_str)

        # 3. Get challenge
        print '\nLogging in with mnemonic passphrase, requesting challenge'
        d =  self.call('gait:login/get_challenge', self.wallet.bitcoin_address())
        d.addCallback(_onChallenge)
        d.addErrback(lambda failure: failure.printTraceback())

    def onLogin(self, login_data):
        last_login = (login_data['last_login']['at'], login_data['last_login']['country'], login_data['last_login']['ip'])
        print '\nLogin successful! Last login on %s, from country %s, ip address: %s' % last_login
        self.createSubaccount()

    def createSubaccount(self):
        def _onSubaccountCreated(data):
            print 'subaccount created: %s' % data
            print '\nQuitting...'
            self.sendClose()

        def _onSubaccountError(failure):
            if str(failure).find('gait:error#alreadyexists') > -1:
                print '\nSubaccount ID or name already existing, please retry'
                self.createSubaccount()

        subaccount_id = raw_input('Please enter your subaccount id (number) or [q] to Quit: ')
        if not subaccount_id.isdigit():
            if subaccount_id.lower() == 'q':
                print '\nQuitting...'
                self.sendClose()
                return False
            print '\nYou must enter a number as subaccount id'
            self.createSubaccount()

        subaccount_id = int(subaccount_id)
        subaccount_name = raw_input('Please enter your subaccount name: ')
        subwallet = self.wallet.subkey(subaccount_id)
        chain = binascii.hexlify(subwallet.chain_code)
        key = Key(subwallet)
        pubkey = binascii.hexlify(key.sec())
        d = self.call('gait:txs/create_subaccount', subaccount_id, subaccount_name, pubkey, chain)
        d.addCallback(_onSubaccountCreated)
        d.addErrback(_onSubaccountError)

if __name__ == '__main__':
    token = requests.get('https://greenaddress.it/token/').text
    # Fetch token from https://test.greenaddress.it for Testnet

    factory = WampClientFactory("wss://prodwss.greenaddress.it/ws/inv", debugWamp=True)
    # Connect to testwss.greenaddress.it for Testnet

    factory.protocol = lambda: GreenAddressClientProtocol(token)
    connectWS(factory)
    reactor.run()
